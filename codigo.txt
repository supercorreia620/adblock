// ---- GET: list categories ----
if ($_SERVER['REQUEST_METHOD'] === 'GET') {
    $stmt = $db->query("SELECT * FROM categories ORDER BY sort_order ASC, id ASC");
    $cats = $stmt->fetchAll();

    $cats = array_map(function ($row) {
        $row['id'] = (int) $row['id'];
        $row['sort_order'] = (int) $row['sort_order'];
        return $row;
    }, $cats);

    echo json_encode(['categories' => $cats]);
    exit;

// ---- POST: add / edit / delete ----
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $input  = json_decode(file_get_contents('php://input'), true);
    $action = $input['action'] ?? '';

    switch ($action) {
        case 'add':
            $name  = trim($input['name'] ?? '');
            $color = trim($input['color'] ?? '#64748B');
            if ($name === '') {
                echo json_encode(['success' => false, 'error' => 'Nome obrigatório']);
                exit;
            }

            $stmt = $db->prepare("INSERT INTO categories (name, color, sort_order) VALUES (:n, :c, (SELECT COALESCE(MAX(sort_order),0)+1 FROM categories))");
            $stmt->execute([':n' => $name, ':c' => $color]);
            echo json_encode(['success' => true, 'id' => (int) $db->lastInsertId()]);
            break;

        case 'edit':
            $id    = (int) ($input['id'] ?? 0);
            $name  = trim($input['name'] ?? '');
            $color = trim($input['color'] ?? '#64748B');
            if ($name === '') {
                echo json_encode(['success' => false, 'error' => 'Nome obrigatório']);
                exit;
            }

            $stmt = $db->prepare("UPDATE categories SET name = :n, color = :c WHERE id = :id");
            $stmt->execute([':n' => $name, ':c' => $color, ':id' => $id]);
            echo json_encode(['success' => true]);
            break;

        case 'delete':
            $id   = (int) ($input['id'] ?? 0);
            $stmt = $db->prepare("DELETE FROM categories WHERE id = :id");
            $stmt->execute([':id' => $id]);
            echo json_encode(['success' => true]);
            break;

        default:
            echo json_encode(['success' => false, 'error' => 'Ação inválida']);
    }
    exit;

// ---- GET: list recurring tasks ----

    $stmt = $db->query("
        SELECT r.*, c.name AS cat_name, c.color AS cat_color, g.name AS group_name
        FROM recurring_tasks r
        LEFT JOIN categories c ON r.category_id = c.id
        LEFT JOIN task_groups g ON r.group_id = g.id
        ORDER BY r.id ASC
    ");
    $items = $stmt->fetchAll();
    $items = array_map(function ($r) {
        $r['id']          = (int)$r['id'];
        $r['category_id'] = $r['category_id'] ? (int)$r['category_id'] : null;
        $r['group_id']    = $r['group_id'] ? (int)$r['group_id'] : null;
        $r['active']      = (bool)$r['active'];
        return $r;
    }, $items);
    echo json_encode(['recurring' => $items]);
    exit;
}

// ---- POST ----
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $input  = json_decode(file_get_contents('php://input'), true);
    $action = $input['action'] ?? '';

    switch ($action) {
        case 'add':
            $title    = trim($input['title'] ?? '');

            $priority = in_array($input['priority'] ?? '', ['high','normal','low']) ? $input['priority'] : 'normal';

            if ($title === '') { echo json_encode(['success' => false, 'error' => 'Título obrigatório']); exit; }

            // Validate days format
            $days = implode(',', array_filter(array_map('intval', explode(',', $days))));
            if ($days === '') $days = '1,2,3,4,5';

            $stmt = $db->prepare("INSERT INTO recurring_tasks (title, category_id, group_id, priority, scheduled_time, days_of_week) VALUES (:t,:c,:g,:p,:s,:d)");

            echo json_encode(['success' => true, 'id' => (int)$db->lastInsertId()]);
            break;

        case 'edit':
            $id       = (int)($input['id'] ?? 0);
            $title    = trim($input['title'] ?? '');
            $catId    = array_key_exists('category_id', $input) ? (!empty($input['category_id']) ? (int)$input['category_id'] : null) : null;
            $groupId  = array_key_exists('group_id', $input) ? (!empty($input['group_id']) ? (int)$input['group_id'] : null) : null;

<?php

header('Content-Type: application/json');

require_once __DIR__ . '/../config/database.php';

$db   = getDB();
$date = $_GET['date'] ?? today();

// Validate date format
if (!preg_match('/^\d{4}-\d{2}-\d{2}$/', $date)) {
    $date = today();
}

// ---- GET: list tasks for a date ----
if ($_SERVER['REQUEST_METHOD'] === 'GET') {
    // Auto-generate recurring tasks for this date
    generateRecurring($db, $date);

    $stmt = $db->prepare("
        SELECT t.*, c.name AS cat_name, c.color AS cat_color,
               g.name AS group_name, g.sort_order AS group_order,
               CASE t.priority 
                   WHEN 'high'   THEN 1 
                   WHEN 'normal' THEN 2 
                   WHEN 'low'    THEN 3 
                   ELSE 2 
               END AS pri_order
        FROM tasks t
        LEFT JOIN categories c ON t.category_id = c.id
        LEFT JOIN task_groups g ON t.group_id = g.id
        WHERE t.task_date = :d
        ORDER BY t.done ASC, g.sort_order IS NULL ASC, g.sort_order ASC, pri_order ASC, t.scheduled_time IS NULL ASC, t.scheduled_time ASC, t.id ASC
    ");
    $stmt->execute([':d' => $date]);
    $tasks = $stmt->fetchAll();

    $tasks = array_map(function ($row) {
        $row['id']             = (int) $row['id'];
        $row['done']           = (bool) $row['done'];
        $row['category_id']    = $row['category_id'] ? (int) $row['category_id'] : null;
        $row['group_id']       = $row['group_id'] ? (int) $row['group_id'] : null;
        $row['group_name']     = $row['group_name'] ?? null;
        $row['priority']       = $row['priority'] ?? 'normal';
        $row['scheduled_time'] = $row['scheduled_time'] ?? null;
        $row['done_by']        = $row['done_by'] ?? null;
        $row['done_at']        = $row['done_at'] ?? null;
        unset($row['pri_order'], $row['group_order']);
        return $row;
    }, $tasks);

    $total = count($tasks);
    $done  = count(array_filter($tasks, fn($t) => $t['done']));

    echo json_encode([
        'tasks'     => $tasks,
        'total'     => $total,
        'done'      => $done,
        'date'      => $date,
        'has_prev'  => hasTasks($db, prevDate($date)),
    ]);
    exit;
}

// ---- POST: add / toggle / delete / copy ----
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $input  = json_decode(file_get_contents('php://input'), true);
    $action = $input['action'] ?? '';

    switch ($action) {
        case 'add':
            $title    = trim($input['title'] ?? '');
            $catId    = isset($input['category_id']) && $input['category_id'] !== '' ? (int) $input['category_id'] : null;
            $groupId  = isset($input['group_id']) && $input['group_id'] !== '' ? (int) $input['group_id'] : null;
            $priority = in_array($input['priority'] ?? '', ['high','normal','low']) ? $input['priority'] : 'normal';
            $sched    = !empty($input['scheduled_time']) ? $input['scheduled_time'] : null;

            if ($title === '') {
                echo json_encode(['success' => false, 'error' => 'Título obrigatório']);
                exit;
            }

            $stmt = $db->prepare(
                "INSERT INTO tasks (title, task_date, category_id, group_id, priority, scheduled_time) VALUES (:t, :d, :c, :g, :p, :s)"
            );
            $stmt->execute([':t' => $title, ':d' => $date, ':c' => $catId, ':g' => $groupId, ':p' => $priority, ':s' => $sched]);
            echo json_encode(['success' => true, 'id' => (int) $db->lastInsertId()]);
            break;

        case 'toggle':
            $id     = (int) ($input['id'] ?? 0);
            $worker = trim($input['worker'] ?? '');

            // Get current state
            $cur = $db->prepare("SELECT done FROM tasks WHERE id = :id");
            $cur->execute([':id' => $id]);
            $row = $cur->fetch();
            if (!$row) {
                echo json_encode(['success' => false, 'error' => 'Tarefa não encontrada']);
                exit;
            }

            $newDone = $row['done'] ? 0 : 1;
            $now     = date('Y-m-d H:i:s');

            if ($newDone) {
                // Marking done
                $stmt = $db->prepare("UPDATE tasks SET done = 1, done_by = :w, done_at = :at WHERE id = :id");
                $stmt->execute([':w' => $worker ?: null, ':at' => $now, ':id' => $id]);

                // Log
                $log = $db->prepare("INSERT INTO task_logs (task_id, action, worker) VALUES (:tid, 'done', :w)");
                $log->execute([':tid' => $id, ':w' => $worker ?: null]);
            } else {
                // Undoing
                $stmt = $db->prepare("UPDATE tasks SET done = 0, done_by = NULL, done_at = NULL WHERE id = :id");
                $stmt->execute([':id' => $id]);

                $log = $db->prepare("INSERT INTO task_logs (task_id, action, worker) VALUES (:tid, 'undone', :w)");
                $log->execute([':tid' => $id, ':w' => $worker ?: null]);
            }

            echo json_encode(['success' => true]);
            break;

        case 'logs':
            $id = (int) ($input['id'] ?? 0);
            $stmt = $db->prepare("SELECT * FROM task_logs WHERE task_id = :id ORDER BY created_at DESC LIMIT 20");
            $stmt->execute([':id' => $id]);
            $logs = $stmt->fetchAll();
            echo json_encode(['success' => true, 'logs' => $logs]);
            break;

        case 'delete':
            $id   = (int) ($input['id'] ?? 0);
            $stmt = $db->prepare("DELETE FROM tasks WHERE id = :id");
            $stmt->execute([':id' => $id]);
            echo json_encode(['success' => true]);
            break;

        case 'copy':
            $from = $input['from_date'] ?? prevDate($date);

            // ---- snip: same copy logic ----
            if (!preg_match('/^\d{4}-\d{2}-\d{2}$/', $from)) {
                echo json_encode(['success' => false, 'error' => 'Data inválida']);
                exit;
            }

            $stmt = $db->prepare("SELECT title, category_id, group_id, priority, scheduled_time FROM tasks WHERE task_date = :d ORDER BY id ASC");
            $stmt->execute([':d' => $from]);
            $prev = $stmt->fetchAll();

            $ins = $db->prepare("INSERT INTO tasks (title, task_date, category_id, group_id, priority, scheduled_time) VALUES (:t, :d, :c, :g, :p, :s)");
            $count = 0;
            foreach ($prev as $row) {
                $chk = $db->prepare("SELECT COUNT(*) FROM tasks WHERE title = :t AND task_date = :d");
                $chk->execute([':t' => $row['title'], ':d' => $date]);
                if ((int) $chk->fetchColumn() === 0) {
                    $ins->execute([
                        ':t' => $row['title'], ':d' => $date, ':c' => $row['category_id'],
                        ':g' => $row['group_id'], ':p' => $row['priority'] ?? 'normal', ':s' => $row['scheduled_time']
                    ]);
                    $count++;
                }
            }

            echo json_encode(['success' => true, 'copied' => $count]);
            break;

        case 'edit':
            $id    = (int) ($input['id'] ?? 0);
            $title = trim($input['title'] ?? '');

            if ($title === '') {
                echo json_encode(['success' => false, 'error' => 'Título obrigatório']);
                exit;
            }

            $fields = ['title = :t'];
            $params = [':t' => $title, ':id' => $id];

            if (array_key_exists('category_id', $input)) {
                $fields[] = 'category_id = :c';
                $params[':c'] = $input['category_id'] !== null && $input['category_id'] !== '' ? (int) $input['category_id'] : null;
            }
            if (array_key_exists('priority', $input)) {
                $fields[] = 'priority = :p';
                $params[':p'] = in_array($input['priority'], ['high','normal','low']) ? $input['priority'] : 'normal';
            }
            if (array_key_exists('scheduled_time', $input)) {
                $fields[] = 'scheduled_time = :s';
                $params[':s'] = !empty($input['scheduled_time']) ? $input['scheduled_time'] : null;
            }
            if (array_key_exists('group_id', $input)) {
                $fields[] = 'group_id = :g';
                $params[':g'] = $input['group_id'] !== null && $input['group_id'] !== '' ? (int) $input['group_id'] : null;
            }

            $sql = "UPDATE tasks SET " . implode(', ', $fields) . " WHERE id = :id";
            $stmt = $db->prepare($sql);
            $stmt->execute($params);
            echo json_encode(['success' => true]);
            break;

        case 'clear_done':
            $stmt = $db->prepare("DELETE FROM tasks WHERE task_date = :d AND done = 1");
            $stmt->execute([':d' => $date]);
            $removed = $stmt->rowCount();
            echo json_encode(['success' => true, 'removed' => $removed]);
            break;

        default:
            echo json_encode(['success' => false, 'error' => 'Ação inválida']);
    }
    exit;
}

// ---- Helpers ----
function prevDate(string $date): string
{
    return date('Y-m-d', strtotime("$date -1 day"));
}

function hasTasks(PDO $db, string $date): bool
{
    $stmt = $db->prepare("SELECT COUNT(*) FROM tasks WHERE task_date = :d");
    $stmt->execute([':d' => $date]);
    return (int) $stmt->fetchColumn() > 0;
}

function generateRecurring(PDO $db, string $date): void
{
    // Day of week: 0=Sun, 1=Mon ... 6=Sat
    $dow = (int) date('w', strtotime($date));

    $stmt = $db->query("SELECT * FROM recurring_tasks WHERE active = 1");
    $recurrings = $stmt->fetchAll();

    $ins = $db->prepare("INSERT INTO tasks (title, task_date, category_id, group_id, priority, scheduled_time, recurring_id) VALUES (:t,:d,:c,:g,:p,:s,:r)");

    foreach ($recurrings as $r) {
        $days = array_map('intval', explode(',', $r['days_of_week']));
        if (!in_array($dow, $days)) continue;

        // Check if already generated
        $chk = $db->prepare("SELECT COUNT(*) FROM tasks WHERE recurring_id = :rid AND task_date = :d");
        $chk->execute([':rid' => $r['id'], ':d' => $date]);
        if ((int) $chk->fetchColumn() > 0) continue;

        $ins->execute([
            ':t' => $r['title'],
            ':d' => $date,
            ':c' => $r['category_id'],
            ':g' => $r['group_id'],
            ':p' => $r['priority'],
            ':s' => $r['scheduled_time'],
            ':r' => $r['id'],
        ]);
    }
}

function buildTimePicker() {
    const hEl = $('tp-hours'), mEl = $('tp-minutes');
    let hHtml = '', mHtml = '';
    for (let i = 0; i < 24; i++) hHtml += `<div class="tp-item" data-val="${i}" onclick="selectHour(${i})">${String(i).padStart(2,'0')}</div>`;
    for (let i = 0; i < 60; i += 5) mHtml += `<div class="tp-item" data-val="${i}" onclick="selectMinute(${i})">${String(i).padStart(2,'0')}</div>`;
    hEl.innerHTML = hHtml;
    mEl.innerHTML = mHtml;
}

function openTimePicker(trigger) {
    activeTimeTarget = trigger;
    const cur = trigger.dataset.value || '';
    if (cur) {
        const [h, m] = cur.split(':').map(Number);
        tpSelH = h; tpSelM = Math.round(m / 5) * 5;
    } else {
        tpSelH = 7; tpSelM = 0;
    }
    updatePickerUI();
    $('modal-time-picker').classList.add('open');
    // Scroll to selected after modal opens
    setTimeout(() => {
        scrollToSelected($('tp-hours'), tpSelH);
        scrollToSelected($('tp-minutes'), tpSelM / 5);
    }, 100);
}

function selectHour(h) { tpSelH = h; updatePickerUI(); }
function selectMinute(m) { tpSelM = m; updatePickerUI(); }

function updatePickerUI() {
    $('tp-display').textContent = `${String(tpSelH).padStart(2,'0')}:${String(tpSelM).padStart(2,'0')}`;

    $('tp-hours').querySelectorAll('.tp-item').forEach(el => {
        el.classList.toggle('selected', parseInt(el.dataset.val) === tpSelH);
    });
    $('tp-minutes').querySelectorAll('.tp-item').forEach(el => {
        el.classList.toggle('selected', parseInt(el.dataset.val) === tpSelM);
    });
}

function scrollToSelected(container, index) {
    const item = container.children[index];
    if (item) {
        item.scrollIntoView({ block: 'center', behavior: 'smooth' });
    }
}

function confirmTimePicker() {
    if (!activeTimeTarget) return;
    const val = `${String(tpSelH).padStart(2,'0')}:${String(tpSelM).padStart(2,'0')}`;
    activeTimeTarget.dataset.value = val;
    activeTimeTarget.querySelector('.time-trigger-text').textContent = `🕐 ${val}`;
    activeTimeTarget.classList.add('has-value');
    closeTimePicker();
}

function clearTimePicker() {
    if (!activeTimeTarget) return;
    activeTimeTarget.dataset.value = '';
    activeTimeTarget.querySelector('.time-trigger-text').textContent = '🕐 Horário';
    activeTimeTarget.classList.remove('has-value');
    closeTimePicker();
}

function closeTimePicker() {
    $('modal-time-picker').classList.remove('open');
    activeTimeTarget = null;
}

function getTimeVal(el) {
    return el.dataset.value || null;
}

function setTimeBtn(el, val) {
    if (val) {
        el.dataset.value = val;
        el.querySelector('.time-trigger-text').textContent = `🕐 ${val}`;
        el.classList.add('has-value');
    } else {
        el.dataset.value = '';
        el.querySelector('.time-trigger-text').textContent = '🕐 Horário';
        el.classList.remove('has-value');
    }
}

    const badge = $('date-badge');

    else if(currentDate===offsetDate(t,-1)){badge.textContent='Ontem';badge.classList.add('show');}
    else if(currentDate===offsetDate(t,1)){badge.textContent='Amanhã';badge.classList.add('show');}
    else{badge.classList.remove('show');}
    $('btn-next').disabled = daysDiff(t,currentDate)>=7;
}

function renderFilterBar(tasks) {
    if(!tasks.length){$('filter-bar').innerHTML='';return;}
    const m=new Map();
    tasks.forEach(t=>{if(t.category_id&&t.cat_name)m.set(t.category_id,{name:t.cat_name,color:t.cat_color});});
    if(!m.size){$('filter-bar').innerHTML='';return;}
    let h=`<button class="filter-chip ${activeFilter===null?'active':''}" onclick="setFilter(null)">Todos</button>`;
    m.forEach((c,id)=>{h+=`<button class="filter-chip ${activeFilter===id?'active':''}" style="--chip-color:${c.color}" onclick="setFilter(${id})">${c.name}</button>`;});

function renderList(tasks, done, total) {
    const filtered = activeFilter!==null ? tasks.filter(t=>t.category_id===activeFilter) : tasks;
    const el = $('task-list');
    if(!tasks.length){el.innerHTML=`<div class="empty-state"><div class="empty-icon">📋</div><p>Nenhuma tarefa neste dia.<br>Adicione acima ou copie do dia anterior!</p></div>`;return;}
    if(!filtered.length){el.innerHTML=`<div class="empty-state"><div class="empty-icon">🔍</div><p>Nenhuma tarefa neste setor.</p></div>`;return;}

    let html = '';
    if(done===total&&total>0&&activeFilter===null) html+=`<div class="all-done"><div class="done-icon">🎉</div><p>Todas concluídas!</p></div>`;

    const grouped = new Map();
    filtered.forEach(t => { const k=t.group_id||0; if(!grouped.has(k))grouped.set(k,[]); grouped.get(k).push(t); });
    const sortedKeys = [...grouped.keys()].sort((a,b) => {
        if(a===0)return 1;if(b===0)return -1;
        return (getGroup(a)?.sort_order||99)-(getGroup(b)?.sort_order||99);
    });

    sortedKeys.forEach(key => {
        const items = grouped.get(key);
        if(key!==0){const g=getGroup(key);const gd=items.filter(t=>t.done).length;
            html+=`<div class="group-header"><span class="group-label">📁 ${g?esc(g.name):'Grupo'}</span><span class="group-line"></span><span class="group-count">${gd}/${items.length}</span></div>`;}
        html += items.map(t=>renderTaskCard(t)).join('');
    });
    el.innerHTML = html;
}

function renderTaskCard(t) {
    const cat=getCat(t.category_id),priClass=t.priority!=='normal'?`pri-${t.priority}`:'';
    let meta='';
    if(cat)meta+=`<span class="cat-badge" style="background:${cat.color}20;color:${cat.color}">${cat.name}</span>`;
    if(t.priority==='high')meta+=`<span class="pri-badge high">Alta</span>`;
    if(t.priority==='low')meta+=`<span class="pri-badge low">Baixa</span>`;
    if(t.scheduled_time)meta+=`<span class="time-badge">🕐 ${formatTime(t.scheduled_time)}</span>`;
    if(t.done&&t.done_by)meta+=`<span class="done-info">✓ ${esc(t.done_by)} · ${formatTimeFromDT(t.done_at)}</span>`;
    else if(t.done&&t.done_at)meta+=`<span class="done-info">✓ ${formatTimeFromDT(t.done_at)}</span>`;

    const td=encodeURIComponent(JSON.stringify({title:t.title,category_id:t.category_id,group_id:t.group_id,priority:t.priority,scheduled_time:t.scheduled_time}));

    return `<li class="task-card ${t.done?'done':''} ${priClass}" data-id="${t.id}">
        <label class="task-check" onclick="event.preventDefault();toggleTask(${t.id},${t.done})">
            <input type="checkbox" ${t.done?'checked':''} tabindex="-1"><span class="mark"></span></label>
        <div class="task-title"><span class="task-name">${esc(t.title)}</span>${meta?`<div class="task-meta">${meta}</div>`:''}</div>
        <div class="task-actions">
            <button class="task-btn log" onclick="viewLogs(${t.id},'${esc(t.title).replace(/'/g,"\\'")}')" title="Histórico">📋</button>
            <button class="task-btn edit" onclick="startEdit(${t.id},JSON.parse(decodeURIComponent('${td}')))" title="Editar">✎</button>

<?php

function getDB(): PDO
{
    static $pdo = null;

    if ($pdo === null) {
        $dbPath = __DIR__ . '/../data/checklist.db';
        $dir    = dirname($dbPath);

        if (!is_dir($dir)) {
            mkdir($dir, 0755, true);
        }

        $pdo = new PDO("sqlite:$dbPath", null, null, [
            PDO::ATTR_ERRMODE            => PDO::ERRMODE_EXCEPTION,
            PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
        ]);

        $pdo->exec("PRAGMA journal_mode=WAL");
        $pdo->exec("PRAGMA foreign_keys=ON");

        migrate($pdo);
    }

    return $pdo;
}

function migrate(PDO $pdo): void
{
    // Categories
    $pdo->exec("
        CREATE TABLE IF NOT EXISTS categories (
            id    INTEGER PRIMARY KEY AUTOINCREMENT,
            name  TEXT NOT NULL UNIQUE,
            color TEXT NOT NULL DEFAULT '#64748B',
            sort_order INTEGER NOT NULL DEFAULT 0
        )
    ");

    // Tasks
    $pdo->exec("
        CREATE TABLE IF NOT EXISTS tasks (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            title       TEXT    NOT NULL,
            done        INTEGER NOT NULL DEFAULT 0,
            task_date   TEXT    NOT NULL,
            category_id INTEGER DEFAULT NULL,
            created_at  TEXT    NOT NULL DEFAULT (datetime('now','localtime')),
            FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE SET NULL
        )
    ");

    // Add category_id if missing (upgrade from v1-v3)
    $cols = $pdo->query("PRAGMA table_info(tasks)")->fetchAll();
    $colNames = array_column($cols, 'name');

    if (!in_array('category_id', $colNames)) {
        $pdo->exec("ALTER TABLE tasks ADD COLUMN category_id INTEGER DEFAULT NULL REFERENCES categories(id) ON DELETE SET NULL");
    }
    if (!in_array('priority', $colNames)) {
        $pdo->exec("ALTER TABLE tasks ADD COLUMN priority TEXT NOT NULL DEFAULT 'normal'");
    }
    if (!in_array('scheduled_time', $colNames)) {
        $pdo->exec("ALTER TABLE tasks ADD COLUMN scheduled_time TEXT DEFAULT NULL");
    }
    if (!in_array('done_by', $colNames)) {
        $pdo->exec("ALTER TABLE tasks ADD COLUMN done_by TEXT DEFAULT NULL");
    }
    if (!in_array('done_at', $colNames)) {
        $pdo->exec("ALTER TABLE tasks ADD COLUMN done_at TEXT DEFAULT NULL");
    }

    // Workers
    $pdo->exec("
        CREATE TABLE IF NOT EXISTS workers (
            id   INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL UNIQUE
        )
    ");

    // Task logs
    $pdo->exec("
        CREATE TABLE IF NOT EXISTS task_logs (
            id         INTEGER PRIMARY KEY AUTOINCREMENT,
            task_id    INTEGER NOT NULL,
            action     TEXT NOT NULL,
            worker     TEXT DEFAULT NULL,
            created_at TEXT NOT NULL DEFAULT (datetime('now','localtime')),
            FOREIGN KEY (task_id) REFERENCES tasks(id) ON DELETE CASCADE
        )
    ");

    // Task groups
    $pdo->exec("
        CREATE TABLE IF NOT EXISTS task_groups (
            id         INTEGER PRIMARY KEY AUTOINCREMENT,
            name       TEXT NOT NULL,
            sort_order INTEGER NOT NULL DEFAULT 0
        )
    ");

    // Recurring tasks
    $pdo->exec("
        CREATE TABLE IF NOT EXISTS recurring_tasks (
            id             INTEGER PRIMARY KEY AUTOINCREMENT,
            title          TEXT    NOT NULL,
            category_id    INTEGER DEFAULT NULL,
            group_id       INTEGER DEFAULT NULL,
            priority       TEXT    NOT NULL DEFAULT 'normal',
            scheduled_time TEXT    DEFAULT NULL,
            days_of_week   TEXT    NOT NULL DEFAULT '1,2,3,4,5',
            active         INTEGER NOT NULL DEFAULT 1,
            FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE SET NULL,
            FOREIGN KEY (group_id) REFERENCES task_groups(id) ON DELETE SET NULL
        )
    ");

    // group_id on tasks
    if (!in_array('group_id', $colNames)) {
        $pdo->exec("ALTER TABLE tasks ADD COLUMN group_id INTEGER DEFAULT NULL REFERENCES task_groups(id) ON DELETE SET NULL");
    }
    // recurring_id on tasks (to avoid duplicates)
    if (!in_array('recurring_id', $colNames)) {
        $pdo->exec("ALTER TABLE tasks ADD COLUMN recurring_id INTEGER DEFAULT NULL");
    }

    // Seed default categories if empty
    $count = (int) $pdo->query("SELECT COUNT(*) FROM categories")->fetchColumn();
    if ($count === 0) {
        $defaults = [
            ['Padaria',     '#F59E0B', 1],
            ['Açougue',     '#EF4444', 2],
            ['Hortifrúti',  '#22C55E', 3],
            ['Estoque',     '#3B82F6', 4],
            ['Limpeza',     '#8B5CF6', 5],
            ['Caixa',       '#EC4899', 6],
            ['Geral',       '#64748B', 7],
        ];
        $stmt = $pdo->prepare("INSERT INTO categories (name, color, sort_order) VALUES (?, ?, ?)");
        foreach ($defaults as $d) {
            $stmt->execute($d);
        }
    }
}
